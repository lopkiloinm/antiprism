%% This template is a verbose curriculum vitae / resume starter using moderncv.

%--------------------------------------------------------------------------------------------------
% 1) Document class and visual style
%--------------------------------------------------------------------------------------------------
% moderncv is purpose-built for resumes and CVs.
% Class options here:
%   - 11pt: base font size
%   - a4paper: page size (change to letterpaper if needed)
%   - sans: sans-serif body style
\documentclass[11pt,a4paper,sans]{moderncv}

% Common styles: casual, classic, banking, oldstyle, fancy
\moderncvstyle{classic}

% Common colors: blue, orange, green, red, purple, grey, black
\moderncvcolor{blue}

% Adjust page scaling/margins. Smaller scale -> larger margins.
\usepackage[scale=0.75]{geometry}

%--------------------------------------------------------------------------------------------------
% 2) Personal information block
%--------------------------------------------------------------------------------------------------
